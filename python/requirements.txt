# Required system dependencies:
#
# On the HOST (where this script runs):
# - schroot       (manage chroot environments)
# - icoutils      (wrestool, icotool for icon extraction)
#
# Inside the CHROOT (where Wine runs):
# - wine          (run Windows applications)
#
# To install on the host:
# sudo apt install schroot icoutils
#
# To install inside the chroot:
# sudo schroot -c debian-amd64
# apt install wine
# exit
